395 SignalScore getSignalScore(){
396 SignalScore signalInfo;
397 signalInfo.rssi = (duckRadio.getRSSI() - RSSI_MIN)/(RSSI_MAX-RSSI_MIN);
398 signalInfo.snr = (duckRadio.getSNR() - SNR_MIN)/(SNR_MAX-SNR_MIN);
399 signalInfo.signalScore = (signalInfo.rssi + signalInfo.snr) / 2.0f;
400 return signalInfo;
